1124 |     async def _receive_frame(self) -> None:
1125 |         # Read the frame header.
1126 |         data = await self._read_bytes(2)
1127 |         header, mask_payloadlen = struct.unpack("BB", data)
1128 |         is_final_frame = header & self.FIN
1129 |         reserved_bits = header & self.RSV_MASK
1130 |         opcode = header & self.OPCODE_MASK
1131 |         opcode_is_control = opcode & 0x8
1132 |         if self._decompressor is not None and opcode != 0:
1133 |             # Compression flag is present in the first frame's header,
1134 |             # but we can't decompress until we have all the frames of
1135 |             # the message.
1136 |             self._frame_compressed = bool(reserved_bits & self.RSV1)
1137 |             reserved_bits &= ~self.RSV1
1138 |         if reserved_bits:
1139 |             # client is using as-yet-undefined extensions; abort
1140 |             self._abort()
1141 |             return
1142 |         is_masked = bool(mask_payloadlen & 0x80)
1143 |         payloadlen = mask_payloadlen & 0x7F
1144 | 
1145 |         # Parse and validate the length.
1146 |         if opcode_is_control and payloadlen >= 126:
1147 |             # control frames must have payload < 126
1148 |             self._abort()
1149 |             return
1150 |         if payloadlen < 126:
1151 |             self._frame_length = payloadlen
1152 |         elif payloadlen == 126:
1153 |             data = await self._read_bytes(2)
1154 |             payloadlen = struct.unpack("!H", data)[0]
1155 |         elif payloadlen == 127:
1156 |             data = await self._read_bytes(8)
1157 |             payloadlen = struct.unpack("!Q", data)[0]
1158 |         new_len = payloadlen
1159 |         if self._fragmented_message_buffer is not None:
1160 |             new_len += len(self._fragmented_message_buffer)
1161 |         if new_len > self.params.max_message_size:
1162 |             self.close(1009, "message too big")
1163 |             self._abort()
1164 |             return
1165 | 
1166 |         # Read the payload, unmasking if necessary.
1167 |         if is_masked:
1168 |             self._frame_mask = await self._read_bytes(4)
1169 |         data = await self._read_bytes(payloadlen)
1170 |         if is_masked:
1171 |             assert self._frame_mask is not None
1172 |             data = _websocket_mask(self._frame_mask, data)
1173 | 
1174 |         # Decide what to do with this frame.
1175 |         if opcode_is_control:
1176 |             # control frames may be interleaved with a series of fragmented
1177 |             # data frames, so control frames must not interact with
1178 |             # self._fragmented_*
1179 |             if not is_final_frame:
1180 |                 # control frames must not be fragmented
1181 |                 self._abort()
1182 |                 return
1183 |         elif opcode == 0:  # continuation frame
