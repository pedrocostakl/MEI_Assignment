1069 | def _oauth_signature(consumer_token, method, url, parameters={}, token=None):
1070 |     """Calculates the HMAC-SHA1 OAuth signature for the given request.
1071 | 
1072 |     See http://oauth.net/core/1.0/#signing_process
1073 |     """
1074 |     parts = urlparse.urlparse(url)
1075 |     scheme, netloc, path = parts[:3]
1076 |     normalized_url = scheme.lower() + "://" + netloc.lower() + path
1077 | 
1078 |     base_elems = []
1079 |     base_elems.append(method.upper())
1080 |     base_elems.append(normalized_url)
1081 |     base_elems.append("&".join("%s=%s" % (k, _oauth_escape(str(v)))
1082 |                                for k, v in sorted(parameters.items())))
1083 |     base_string = "&".join(_oauth_escape(e) for e in base_elems)
1084 | 
1085 |     key_elems = [escape.utf8(consumer_token["secret"])]
1086 |     key_elems.append(escape.utf8(token["secret"] if token else ""))
1087 |     key = b"&".join(key_elems)
1088 | 
1089 |     hash = hmac.new(key, escape.utf8(base_string), hashlib.sha1)
1090 |     return binascii.b2a_base64(hash.digest())[:-1]
