 519 |     def run_sync(self, func, timeout=None):
 520 |         """Starts the `IOLoop`, runs the given function, and stops the loop.
 521 | 
 522 |         The function must return either a yieldable object or
 523 |         ``None``. If the function returns a yieldable object, the
 524 |         `IOLoop` will run until the yieldable is resolved (and
 525 |         `run_sync()` will return the yieldable's result). If it raises
 526 |         an exception, the `IOLoop` will stop and the exception will be
 527 |         re-raised to the caller.
 528 | 
 529 |         The keyword-only argument ``timeout`` may be used to set
 530 |         a maximum duration for the function.  If the timeout expires,
 531 |         a `tornado.util.TimeoutError` is raised.
 532 | 
 533 |         This method is useful in conjunction with `tornado.gen.coroutine`
 534 |         to allow asynchronous calls in a ``main()`` function::
 535 | 
 536 |             @gen.coroutine
 537 |             def main():
 538 |                 # do stuff...
 539 | 
 540 |             if __name__ == '__main__':
 541 |                 IOLoop.current().run_sync(main)
 542 | 
 543 |         .. versionchanged:: 4.3
 544 |            Returning a non-``None``, non-yieldable value is now an error.
 545 | 
 546 |         .. versionchanged:: 5.0
 547 |            If a timeout occurs, the ``func`` coroutine will be cancelled.
 548 |         """
 549 |         future_cell = [None]
 550 | 
 551 |         def run():
 552 |             try:
 553 |                 result = func()
 554 |                 if result is not None:
 555 |                     from tornado.gen import convert_yielded
 556 |                     result = convert_yielded(result)
 557 |             except Exception:
 558 |                 future_cell[0] = Future()
 559 |                 future_set_exc_info(future_cell[0], sys.exc_info())
 560 |             else:
 561 |                 if is_future(result):
 562 |                     future_cell[0] = result
 563 |                 else:
 564 |                     future_cell[0] = Future()
 565 |                     future_cell[0].set_result(result)
 566 |             self.add_future(future_cell[0], lambda future: self.stop())
 567 |         self.add_callback(run)
 568 |         if timeout is not None:
 569 |             def timeout_callback():
 570 |                 # If we can cancel the future, do so and wait on it. If not,
 571 |                 # Just stop the loop and return with the task still pending.
 572 |                 # (If we neither cancel nor wait for the task, a warning
 573 |                 # will be logged).
 574 |                 if not future_cell[0].cancel():
 575 |                     self.stop()
 576 |             timeout_handle = self.add_timeout(self.time() + timeout, timeout_callback)
 577 |         self.start()
 578 |         if timeout is not None:
 579 |             self.remove_timeout(timeout_handle)
 580 |         if future_cell[0].cancelled() or not future_cell[0].done():
 581 |             raise TimeoutError('Operation timed out after %s seconds' % timeout)
 582 |         return future_cell[0].result()
