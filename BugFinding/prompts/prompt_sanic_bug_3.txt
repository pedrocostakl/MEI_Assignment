 798 |     def url_for(self, view_name: str, **kwargs):

 800 |

 803 |         specified parameter type. If these conditions are not met, a
 804 |         `URLBuildError` will be thrown.
 805 | 
 806 |         Keyword arguments that are not request parameters will be included in
 807 |         the output URL's query string.
 808 | 
 809 |         :param view_name: string referencing the view name
 810 |         :param \**kwargs: keys and values that are used to build request
 811 |             parameters and query string arguments.
 812 | 
 813 |         :return: the built URL
 814 | 
 815 |         Raises:
 816 |             URLBuildError
 817 |         """
 818 |         # find the route by the supplied view name
 819 |         kw: Dict[str, str] = {}
 820 |         # special static files url_for
 821 |         if view_name == "static":
 822 |             kw.update(name=kwargs.pop("name", "static"))
 823 |         elif view_name.endswith(".static"):  # blueprint.static
 824 |             kwargs.pop("name", None)
 825 |             kw.update(name=view_name)
 826 | 
 827 |         uri, route = self.router.find_route_by_view_name(view_name, **kw)
 828 |         if not (uri and route):

 831 |             )
 832 | 
 833 |         if view_name == "static" or view_name.endswith(".static"):
 834 |             filename = kwargs.pop("filename", None)
 835 |             # it's static folder
 836 |             if "<file_uri:" in uri:
 837 |                 folder_ = uri.split("<file_uri:", 1)[0]
 838 |                 if folder_.endswith("/"):
 839 |                     folder_ = folder_[:-1]
 840 | 
 841 |                 if filename.startswith("/"):
 842 |                     filename = filename[1:]
 843 |

 845 | 
 846 |         if uri != "/" and uri.endswith("/"):
 847 |             uri = uri[:-1]
 848 | 
 849 |         out = uri

 851 |         # find all the parameters we will need to build in the URL
 852 |         matched_params = re.findall(self.router.parameter_pattern, uri)
 853 | 
 854 |         # _method is only a placeholder now, don't know how to support it
 855 |         kwargs.pop("_method", None)
 856 |         anchor = kwargs.pop("_anchor", "")
 857 |         # _external need SERVER_NAME in config or pass _server arg
 858 |         external = kwargs.pop("_external", False)
 859 |         scheme = kwargs.pop("_scheme", "")

 861 |             raise ValueError("When specifying _scheme, _external must be True")

 879 |             # we only want to match against each individual parameter
 880 |             specific_pattern = "^{}$".format(pattern)

 885 |                 del kwargs[name]
 886 |             else:
 887 |                 raise URLBuildError(

 889 |                         name

 891 |                 )
