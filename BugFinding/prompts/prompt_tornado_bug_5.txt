 518 |     def run_sync(self, func, timeout=None):
 519 |         """Starts the `IOLoop`, runs the given function, and stops the loop.
 520 | 
 521 |         The function must return either a yieldable object or
 522 |         ``None``. If the function returns a yieldable object, the
 523 |         `IOLoop` will run until the yieldable is resolved (and
 524 |         `run_sync()` will return the yieldable's result). If it raises
 525 |         an exception, the `IOLoop` will stop and the exception will be
 526 |         re-raised to the caller.
 527 | 
 528 |         The keyword-only argument ``timeout`` may be used to set
 529 |         a maximum duration for the function.  If the timeout expires,
 530 |         a `tornado.util.TimeoutError` is raised.
 531 | 
 532 |         This method is useful in conjunction with `tornado.gen.coroutine`
 533 |         to allow asynchronous calls in a ``main()`` function::
 534 | 
 535 |             @gen.coroutine
 536 |             def main():
 537 |                 # do stuff...
 538 | 
 539 |             if __name__ == '__main__':
 540 |                 IOLoop.current().run_sync(main)
 541 | 
 542 |         .. versionchanged:: 4.3
 543 |            Returning a non-``None``, non-yieldable value is now an error.
 544 | 
 545 |         .. versionchanged:: 5.0
 546 |            If a timeout occurs, the ``func`` coroutine will be cancelled.
 547 |         """
 548 |         future_cell = [None]
 549 | 
 550 |         def run():
 551 |             try:
 552 |                 result = func()
 553 |                 if result is not None:
 554 |                     from tornado.gen import convert_yielded
 555 |                     result = convert_yielded(result)
 556 |             except Exception:
 557 |                 future_cell[0] = Future()
 558 |                 future_set_exc_info(future_cell[0], sys.exc_info())
 559 |             else:
 560 |                 if is_future(result):
 561 |                     future_cell[0] = result
 562 |                 else:
 563 |                     future_cell[0] = Future()
 564 |                     future_cell[0].set_result(result)
 565 |             self.add_future(future_cell[0], lambda future: self.stop())
 566 |         self.add_callback(run)
 567 |         if timeout is not None:
 568 |             def timeout_callback():
 569 |                 # If we can cancel the future, do so and wait on it. If not,
 570 |                 # Just stop the loop and return with the task still pending.
 571 |                 # (If we neither cancel nor wait for the task, a warning
 572 |                 # will be logged).
 573 |                 if not future_cell[0].cancel():
 574 |                     self.stop()
 575 |             timeout_handle = self.add_timeout(self.time() + timeout, timeout_callback)
 576 |         self.start()
 577 |         if timeout is not None:
 578 |             self.remove_timeout(timeout_handle)
 579 |         if future_cell[0].cancelled() or not future_cell[0].done():
 580 |             raise TimeoutError('Operation timed out after %s seconds' % timeout)
 581 |         return future_cell[0].result()

 926 |     def start(self):
 927 |         if self._running:
 928 |             raise RuntimeError("IOLoop is already running")
 929 |         if os.getpid() != self._pid:
 930 |             raise RuntimeError("Cannot share PollIOLoops across processes")
 931 |         self._setup_logging()
 932 |         if self._stopped:
 933 |             self._stopped = False
 934 |             return
 935 |         old_current = IOLoop.current(instance=False)
 936 |         if old_current is not self:
 937 |             self.make_current()

 939 |         self._running = True
 940 | 
 941 |         # signal.set_wakeup_fd closes a race condition in event loops:
 942 |         # a signal may arrive at the beginning of select/poll/etc
 943 |         # before it goes into its interruptible sleep, so the signal
 944 |         # will be consumed without waking the select.  The solution is
 945 |         # for the (C, synchronous) signal handler to write to a pipe,
 946 |         # which will then be seen by select.
 947 |         #
 948 |         # In python's signal handling semantics, this only matters on the
 949 |         # main thread (fortunately, set_wakeup_fd only works on the main
 950 |         # thread and will raise a ValueError otherwise).
 951 |         #
 952 |         # If someone has already set a wakeup fd, we don't want to
 953 |         # disturb it.  This is an issue for twisted, which does its
 954 |         # SIGCHLD processing in response to its own wakeup fd being
 955 |         # written to.  As long as the wakeup fd is registered on the IOLoop,
 956 |         # the loop will still wake up and everything should work.
 957 |         old_wakeup_fd = None
 958 |         if hasattr(signal, 'set_wakeup_fd') and os.name == 'posix':
 959 |             # requires python 2.6+, unix.  set_wakeup_fd exists but crashes
 960 |             # the python process on windows.
 961 |             try:
 962 |                 old_wakeup_fd = signal.set_wakeup_fd(self._waker.write_fileno())
 963 |                 if old_wakeup_fd != -1:
 964 |                     # Already set, restore previous value.  This is a little racy,
 965 |                     # but there's no clean get_wakeup_fd and in real use the
 966 |                     # IOLoop is just started once at the beginning.
 967 |                     signal.set_wakeup_fd(old_wakeup_fd)
 968 |                     old_wakeup_fd = None
 969 |             except ValueError:

 971 |                 # is no longer valid.
 972 |                 old_wakeup_fd = None

 976 |                 # Prevent IO event starvation by delaying new callbacks
 977 |                 # to the next iteration of the event loop.

 982 |                 # are ready, so timeouts that call add_timeout cannot
 983 |                 # schedule anything in this iteration.

 996 |                         else:

1001 |                         # more than half cancellations.
1002 |                         self._cancellations = 0

1004 |                                           if x.callback is not None]
1005 |                         heapq.heapify(self._timeouts)

1007 |                 for i in range(ncallbacks):
1008 |                     self._run_callback(self._callbacks.popleft())
1009 |                 for timeout in due_timeouts:
1010 |                     if timeout.callback is not None:
1011 |                         self._run_callback(timeout.callback)

1013 |                 # them to be freed before we go into our poll wait.
1014 |                 due_timeouts = timeout = None

1016 |                 if self._callbacks:

1022 |                     # Use self.time() instead of 'now' to account for time

1034 |                     # clear alarm so it doesn't fire while poll is waiting for
1035 |                     # events.

1038 |                 try:

1042 |                     # different exception types may be thrown and there are
