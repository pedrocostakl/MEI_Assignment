 389 |     def _convert_header_value(self, value: _HeaderTypes) -> str:
 390 |         # Convert the input value to a str. This type check is a bit
 391 |         # subtle: The bytes case only executes on python 3, and the
 392 |         # unicode case only executes on python 2, because the other
 393 |         # cases are covered by the first match for str.
 394 |         if isinstance(value, str):
 395 |             retval = value
 396 |         elif isinstance(value, bytes):  # py3
 397 |             # Non-ascii characters in headers are not well supported,
 398 |             # but if you pass bytes, use latin1 so they pass through as-is.
 399 |             retval = value.decode("latin1")
 400 |         elif isinstance(value, unicode_type):  # py2
 401 |             # TODO: This is inconsistent with the use of latin1 above,
 402 |             # but it's been that way for a long time. Should it change?
 403 |             retval = escape.utf8(value)
 404 |         elif isinstance(value, numbers.Integral):
 405 |             # return immediately since we know the converted value will be safe
 406 |             return str(value)
 407 |         elif isinstance(value, datetime.datetime):
 408 |             return httputil.format_timestamp(value)
 409 |         else:
 410 |             raise TypeError("Unsupported header value %r" % value)
 411 |         # If \n is allowed into the header, it is possible to inject
 412 |         # additional headers or split the request.
 413 |         if RequestHandler._INVALID_HEADER_CHAR_RE.search(retval):

1059 |         if another flush occurs before the previous flush's callback
1060 |         has been run, the previous callback will be discarded.
1061 | 
1062 |         .. versionchanged:: 4.0
1063 |            Now returns a `.Future` if no callback is given.
1064 | 
1065 |         .. versionchanged:: 6.0
1066 | 
1067 |            The ``callback`` argument was removed.
1068 |         """

1070 |         chunk = b"".join(self._write_buffer)
1071 |         self._write_buffer = []

1073 |             self._headers_written = True
1074 |             for transform in self._transforms:
1075 |                 assert chunk is not None
1076 |                 self._status_code, self._headers, chunk = transform.transform_first_chunk(
1077 |                     self._status_code, self._headers, chunk, include_footers

1105 |     def finish(self, chunk: Union[str, bytes, dict] = None) -> "Future[None]":
1106 |         """Finishes this response, ending the HTTP request.
1107 | 
1108 |         Passing a ``chunk`` to ``finish()`` is equivalent to passing that
1109 |         chunk to ``write()`` and then calling ``finish()`` with no arguments.
1110 | 
1111 |         Returns a `.Future` which may optionally be awaited to track the sending
1112 |         of the response to the client. This `.Future` resolves when all the response
1113 |         data has been sent, and raises an error if the connection is closed before all
1114 |         data can be sent.
1115 | 
1116 |         .. versionchanged:: 5.1
1117 | 
1118 |            Now returns a `.Future` instead of ``None``.
1119 |         """
1120 |         if self._finished:
1121 |             raise RuntimeError("finish() called twice")
1122 | 
1123 |         if chunk is not None:
1124 |             self.write(chunk)
1125 | 
1126 |         # Automatically support ETags and add the Content-Length header if
1127 |         # we have not flushed any content yet.
1128 |         if not self._headers_written:
1129 |             if (
1130 |                 self._status_code == 200
1131 |                 and self.request.method in ("GET", "HEAD")
1132 |                 and "Etag" not in self._headers
1133 |             ):
1134 |                 self.set_etag_header()
1135 |                 if self.check_etag_header():
1136 |                     self._write_buffer = []

1140 |             ):

1143 |                 )

1146 |                 content_length = sum(len(part) for part in self._write_buffer)
1147 |                 self.set_header("Content-Length", content_length)

1151 |         # set on the HTTPConnection (which would otherwise prevent the

1153 |         # are keepalive connections)
