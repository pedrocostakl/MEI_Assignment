 361 |     def _convert_header_value(self, value):
 362 |         # type: (_HeaderTypes) -> str
 363 | 
 364 |         # Convert the input value to a str. This type check is a bit
 365 |         # subtle: The bytes case only executes on python 3, and the
 366 |         # unicode case only executes on python 2, because the other
 367 |         # cases are covered by the first match for str.
 368 |         if isinstance(value, str):
 369 |             retval = value
 370 |         elif isinstance(value, bytes):  # py3

 911 |     def flush(self, include_footers=False, callback=None):
 912 |         """Flushes the current output buffer to the network.
 913 | 
 914 |         The ``callback`` argument, if given, can be used for flow control:
 915 |         it will be run when all flushed data has been written to the socket.
 916 |         Note that only one flush callback can be outstanding at a time;
 917 |         if another flush occurs before the previous flush's callback
 918 |         has been run, the previous callback will be discarded.
 919 | 
 920 |         .. versionchanged:: 4.0
 921 |            Now returns a `.Future` if no callback is given.
 922 |         """
 923 |         chunk = b"".join(self._write_buffer)
 924 |         self._write_buffer = []
 925 |         if not self._headers_written:
 926 |             self._headers_written = True
 927 |             for transform in self._transforms:
 928 |                 self._status_code, self._headers, chunk = \
 929 |                     transform.transform_first_chunk(
 930 |                         self._status_code, self._headers,
 931 |                         chunk, include_footers)
 932 |             # Ignore the chunk and only write the headers for HEAD requests
 933 |             if self.request.method == "HEAD":
 934 |                 chunk = None
 935 | 
 936 |             # Finalize the cookie headers (which have been stored in a side
 937 |             # object so an outgoing cookie could be overwritten before it
 938 |             # is sent).
 939 |             if hasattr(self, "_new_cookie"):

 941 |                     self.add_header("Set-Cookie", cookie.OutputString(None))

 943 |             start_line = httputil.ResponseStartLine('',
 944 |                                                     self._status_code,
 945 |                                                     self._reason)
 946 |             return self.request.connection.write_headers(
 947 |                 start_line, self._headers, chunk, callback=callback)
 948 |         else:
 949 |             for transform in self._transforms:
 950 |                 chunk = transform.transform_chunk(chunk, include_footers)
 951 |             # Ignore the chunk and only write the headers for HEAD requests
 952 |             if self.request.method != "HEAD":

 960 |         """Finishes this response, ending the HTTP request."""
 961 |         if self._finished:
 962 |             raise RuntimeError("finish() called twice")
 963 | 
 964 |         if chunk is not None:

 968 |         # we have not flushed any content yet.

 973 |                 self.set_etag_header()

2811 |     # better CPU/size tradeoff.

2814 |     # after considering the "Content-Encoding: gzip" header and the header
2815 |     # inside the gzip encoding.

2817 |     # regardless of size.

2826 |     def transform_first_chunk(self, status_code, headers, chunk, finishing):
2827 |         # type: (int, httputil.HTTPHeaders, bytes, bool) -> typing.Tuple[int, httputil.HTTPHeaders, bytes]
2828 |         # TODO: can/should this type be inherited from the superclass?
2829 |         if 'Vary' in headers:

2831 |         else:

2834 |             ctype = _unicode(headers.get("Content-Type", "")).split(";")[0]

2836 |                 (not finishing or len(chunk) >= self.MIN_LENGTH) and \

 358 |     def check_origin(self, origin):

 363 |         method is not called for clients that do not send this header;

 366 |         clients do not have the same cross-site security concerns).
