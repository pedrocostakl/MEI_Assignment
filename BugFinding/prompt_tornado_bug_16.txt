 215 |             result = func(*args, **kwargs)

 285 | class WaitIterator(object):
 286 |     """Provides an iterator to yield the results of futures as they finish.
 287 | 
 288 |     Yielding a set of futures like this:
 289 | 
 290 |     ``results = yield [future1, future2]``

 292 |     pauses the coroutine until both ``future1`` and ``future2``
 293 |     return, and then restarts the coroutine with the results of both
 294 |     futures. If either future is an exception, the expression will
 295 |     raise that exception and all the results will be lost.
 296 | 
 297 |     If you need to get the result of each future as soon as possible,
 298 |     or if you need the result of some futures even if others produce
 299 |     errors, you can use ``WaitIterator``::
 300 | 
 301 |       wait_iterator = gen.WaitIterator(future1, future2)
 302 |       while not wait_iterator.done():
 303 |           try:
 304 |               result = yield wait_iterator.next()

 306 |               print("Error {} from {}".format(e, wait_iterator.current_future))
 307 |           else:

 311 | 
 312 |     Because results are returned as soon as they are available the
 313 |     output from the iterator *will not be in the same order as the
 314 |     input arguments*. If you need to know which future produced the
 315 |     current result, you can use the attributes
 316 |     ``WaitIterator.current_future``, or ``WaitIterator.current_index``
 317 |     to get the index of the future from the input list. (if keyword
 318 |     arguments were used in the construction of the `WaitIterator`,
 319 |     ``current_index`` will use the corresponding keyword).
 320 | 
 321 |     .. versionadded:: 4.1
 322 |     """
 323 |     def __init__(self, *args, **kwargs):
 324 |         if args and kwargs:
 325 |             raise ValueError(
 326 |                 "You must provide args or kwargs, not both")
 327 | 
 328 |         if kwargs:
 329 |             self._unfinished = dict((f, k) for (k, f) in kwargs.items())
 330 |             futures = list(kwargs.values())
 331 |         else:
 332 |             self._unfinished = dict((f, i) for (i, f) in enumerate(args))

 349 |         self.current_index = self.current_future = None
